// switch
	·与javascript if语法一致
	
	switch(expression)
		·expression 类型 必须是 byte short int char 或枚举 JDK 7 开始 可以为String
		
		int a = 2;
		switch (a) {
		case 2:
			System.out.println(3);
			System.out.println(4);
			break;
			// case 语句中可以没有大括号， break语句是可以选的
		default:
			System.out.println(5);
		}
